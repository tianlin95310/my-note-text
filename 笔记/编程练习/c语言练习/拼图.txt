void shang()
{
	if(i<=0)
		return;
	else
	{
		temp = a[i][j];
		a[i][j] = a[i-1][j];
		a[i-1][j] = temp;
		i = i-1;
	}
	showind();
}
void zuo()
{
	if(j<=0)
		return;
	else
	{
		temp = a[i][j];
		a[i][j] = a[i][j-1];
		a[i][j-1] = temp;
		j = j-1;
	}
	showind();
}
void you()
{
	if(j>=2)
		return;
	else
	{
		temp = a[i][j];
		a[i][j] = a[i][j+1];
		a[i][j+1] = temp;
		j = j+1;
	}
	showind();
}
void xia()
{
	if(i>=2)
		return;
	else
	{
		temp = a[i][j];
		a[i][j] = a[i+1][j];
		a[i+1][j] = temp;
		i = i+1;
	}
	showind();
}
