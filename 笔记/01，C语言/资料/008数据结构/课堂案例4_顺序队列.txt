#include <stdio.h>
#include <string.h>
#include <malloc.h>

#define dim(x) sizeof(x) / sizeof(x[0])

struct SeQueuestr
{
	int elem[100];
	int front;
	int rear;
};

void Add(SeQueuestr *pDeq, int x)
{
	if (pDeq->rear <= 100)
	{
		pDeq->elem[pDeq->rear++] = x;
	}
}

void Del(SeQueuestr *pDeq)
{
	if (pDeq->rear > 0)
	{
		for (int i = pDeq->front; i < pDeq->rear - 1; i++)
		{
			pDeq->elem[i] = pDeq->elem[i+1];
		}
		pDeq->rear--;
	}
}

void Print(SeQueuestr *pDeq)
{
	SeQueuestr *pCurr = pDeq;
	for (int i = pCurr->front; i < pCurr->rear; i++)
	{
		printf("%d ", pCurr->elem[i]);
	}
	printf("\r\n");
}

void main()
{
	SeQueuestr deq;
	deq.front = 0;
	deq.rear = 0;

	Add(&deq, 11);
	Add(&deq, 22);
	Add(&deq, 33);
	Add(&deq, 44);
	Print(&deq);

	Del(&deq);
	Print(&deq);
	Del(&deq);
	Print(&deq);
	Del(&deq);
	Print(&deq);
	Del(&deq);
	Print(&deq);
}
