1， int a[2][3] ={{1,2,3}，{4,5,6}};
	后者的值一定要给出，前者不给值的话，就一定要初始化。
	{}内部的大括号可以不写，
2，
    int m[2];
	m的类型为int (*)[2].
	a指向的是{1,2,3,}中的1，a+1指向的是{4,5,6}中的4,
	而&a的类型是int(*)[2][3]指向的是{{1,2,3}，{4,5,6}}
	整体中的1;&a+1指向的是下一个整体。

  int b[3] = {4,5,6};

  int *c[2] = {a,b};c的类型为int**;a为int*，b为int*
